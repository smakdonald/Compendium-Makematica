The map
\begin{equation*} \xymatrix@R=1mm{ \operatorname{End}_R(M^{\oplus n}) &\ar[l]_-{\zeta} \operatorname{Mat}_n(\operatorname{End}_R(M)) \\ \sum_{i,j} \iota_i \alpha_{i,j} \pi_j & \ar@{|->}[l] [\alpha_{i,j}]_{i,j} } \end{equation*}
is a two-sided inverse for \(\Theta\text{:}\)
\begin{equation*} \zeta (\Theta(\phi)) = \zeta ( [\pi_i \phi \iota_j]_{i,j} ) = \sum_{i,j} \iota_i \pi_i \phi \iota_j \pi_j =(\sum_i \iota_i \pi_i) \phi (\sum_j \iota_j \pi_j)=\phi, \quad\text{and} \end{equation*}
\begin{equation*} \Theta(\zeta([\alpha_{i,j}]_{i,j}]))_{k,\ell} = \Theta ( \sum_{i,j} \iota_i \alpha_{i,j} \pi_j)_{k,\ell} = \pi_k (\sum_{i,j} \iota_i \alpha_{i,j} \pi_j) \iota_\ell = \sum_{i,j} (\pi_k \iota_i) \alpha_{i,j} (\pi_j \iota_\ell) = \alpha_{k,\ell}. \end{equation*}
